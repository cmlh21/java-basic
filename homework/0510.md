## Student 테이블 생성 후 메뉴 만들기

1) createTable() 함수 정의

- Student 테이블 CREATE 
  - no INT PRIMARY KEY AUTO_INCREMENT
  - name VARCHAR(40) NOT NULL
  - 국, 영, 수 INT(3) CHECK 0 이상 100 이하
  - 평균 FLOAT(5, 2)
  - 학점 VARCHAR(1) 기본값 F
  - 등록일자 DATETIME 기본값 현재시간
- 2) 메인 내용
  - 메뉴)
    1. 학생 등록
    2. 모든 학생 보기
    3. 이름으로 검색
    4. 학번으로 검색
    5. 1등 보기 
    6. 종료

package com.day02.quiz;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

import javax.swing.JOptionPane;

public class Quiz02 {
	private static final String JDBC_DRIVER = "org.mariadb.jdbc.Driver";
	private static final String DB_URL = "jdbc:mysql://localhost:3306/testDB?"+ "useUnicode=true"+ "&characterEncoding=utf8";
	private static final String USERNAME = "root";
	private static final String PASSWORD = "1234";
	
	public void createTable() {
		Connection conn = null;
		PreparedStatement ps = null;
		
		try {
			Class.forName(JDBC_DRIVER);
			conn = DriverManager.getConnection(DB_URL, USERNAME, PASSWORD);
			String sql = "CREATE TABLE student(no INT PRIMARY KEY AUTO_INCREMENT, "
					+ "name VARCHAR(40) NOT NULL, "
					+ "korean INT(3) CHECK(korean >= 0 AND korean <= 100), "
					+ "english INT(3) CHECK(english >= 0 AND english <= 100), "
					+ "math INT(3) CHECK(math >= 0 AND math <= 100), "
					+ "avg FLOAT(5, 2), "
					+ "grade VARCHAR(1) DEFAULT 'F', "
					+ "regdate DATETIME DEFAULT CURRENT_TIMESTAMP)";
			ps = conn.prepareStatement(sql);
			ps.execute();
			
		} catch(Exception e){
			e.printStackTrace();
		} finally {
			try {
				if(ps != null) {ps.close();}
				if(conn != null) {conn.close();}
			} catch(Exception e) {
				e.printStackTrace();
			}
		}
	}
	
	
	public static void main(String[] args) {
		
		Connection conn = null;
		PreparedStatement ps = null;
		ResultSet rs = null;
		
		Quiz02 ct = new Quiz02();
//		ct.createTable();
		
		String name = null;
		int ko = 0;
		int en = 0;
		int ma = 0;
		
		String menu = "1. 학생 등록 \n"
					+ "2. 모든 학생 보기 \n"
					+ "3. 이름으로 검색 \n"
					+ "4. 학번으로 검색 \n"
					+ "5. 1등 보기 \n"
					+ "0. 종료";
		String select;
		while(true) {
			select = JOptionPane.showInputDialog(menu);
			if("1".equals(select)) { // 학생 등록
				name = JOptionPane.showInputDialog("새 학생이름 입력");
				ko = Integer.parseInt(JOptionPane.showInputDialog("국어 점수 입력"));
				en = Integer.parseInt(JOptionPane.showInputDialog("영어 점수 입력"));
				ma = Integer.parseInt(JOptionPane.showInputDialog("수학 점수 입력"));
				Double avg = (ko + en + ma) / 3.0;
				String grade = "F";
				if(avg >= 90) {
					grade = "A";
				} else if(avg >= 80) {
					grade = "B";
				} else if(avg >= 70) {
					grade = "C";
				} else if(avg >= 60) {
					grade = "D";
				}
				
				try {
					Class.forName(JDBC_DRIVER);
					conn = DriverManager.getConnection(DB_URL, USERNAME, PASSWORD);
					
					String sql = "INSERT INTO student VALUES(NULL, ?, ?, ?, ?, ?, ?, NULL)";
					ps = conn.prepareStatement(sql);
					ps.setString(1, name);
					ps.setInt(2, ko);
					ps.setInt(3, en);
					ps.setInt(4, ma);
					ps.setDouble(5, avg);
					ps.setString(6, grade);
					ps.execute();
				} catch (Exception e) {
					e.printStackTrace();
				} 
			} else if("2".equals(select)) { // 모든 학생 보기
				try {
					Class.forName(JDBC_DRIVER);
					conn = DriverManager.getConnection(DB_URL, USERNAME, PASSWORD);
					String sql = "SELECT * FROM student";
					ps = conn.prepareStatement(sql);
					rs = ps.executeQuery();
					
					while(rs.next()) {
						JOptionPane.showMessageDialog(null, rs.getInt(1) + "\n"
										  + rs.getInt(2) + "\n"
										  + rs.getInt(3) + "\n"
										  + rs.getInt(4) + "\n"
										  + rs.getInt(5) + "\n"
										  + rs.getInt(6) + "\n"
										  + rs.getInt(7) + "\n"
										  + rs.getInt(8) + "\n"
										  + "--------------" + "\n");
					}
					
				} catch(Exception e) {
					e.printStackTrace();
				}
			} else if("3".equals(select)) { // 이름으로 검색
				try {
					Class.forName(JDBC_DRIVER);
					conn = DriverManager.getConnection(DB_URL, USERNAME, PASSWORD);
					String searchName = JOptionPane.showInputDialog("찾을 학생 이름 입력");
					String sql = "SELECT * FROM student WHERE name = '" + searchName + "'";
					ps = conn.prepareStatement(sql);
					ps.execute();
				} catch(Exception e) {
					e.printStackTrace();
				}
				
			} else if("4".equals(select)) { // 학번으로 검색
				try {
					Class.forName(JDBC_DRIVER);
					conn = DriverManager.getConnection(DB_URL, USERNAME, PASSWORD);
					String searchNo = JOptionPane.showInputDialog("찾을 학생 번호 입력");
					String sql = "SELECT * FROM student WHERE no = '" + searchNo + "'";
					ps = conn.prepareStatement(sql);
					ps.execute();
				} catch(Exception e) {
					e.printStackTrace();
				}
				
			} else if("5".equals(select)) { // 1등 보기
				try {
					Class.forName(JDBC_DRIVER);
					conn = DriverManager.getConnection(DB_URL, USERNAME, PASSWORD);
					String sql = "SELECT * FROM student WHERE avg = (SELECT MAX(avg) FROM student)";
					ps = conn.prepareStatement(sql);
					ps.execute();
				} catch(Exception e) {
					e.printStackTrace();
				}
				
			} else if("0".equals(select)) { // 종료
				JOptionPane.showMessageDialog(null,"프로그램을 종료합니다.");
				break;
			} else {
				JOptionPane.showMessageDialog(null, "다시 입력하세요");
			}
		}
	}
}
